                           <label className="block text-xs font-bold text-indigo-800 uppercase mb-1">Pilih Ebook ke-Berapa</label>
                           <select name="selectedEbookId" value={taskConfig.selectedEbookId} onChange={handleTaskConfigChange} className="w-full rounded border-indigo-200 p-2 text-sm mb-2 bg-white">
                              {(EBOOK_SERIES_DATA[projectData.level] || []).map((ebook) => (
                                 <option key={ebook.id} value={ebook.id}>{ebook.label}</option>
                              ))}
                           </select>
                        </div>
                    )}

                    {/* DROPDOWN CHAPTER SELECTOR */}
                    <div>
                      <label className="block text-xs font-bold text-indigo-800 uppercase mb-1">Pilih Bab / Tulis Manual</label>
                      <select name="judulBab" value={taskConfig.judulBab} onChange={handleTaskConfigChange} className="w-full rounded border-indigo-200 p-2 text-sm mb-2 bg-white">
                        <option value="">-- Pilih Bab untuk Ditulis --</option>
                        {CHAPTER_TEMPLATES.map((chapter, idx) => (<option key={idx} value={chapter}>{chapter}</option>))}
                      </select>
                      {taskConfig.judulBab === "Custom / Tulis Judul Sendiri..." && (
                        <input type="text" name="manualJudulBab" value={taskConfig.manualJudulBab} onChange={handleTaskConfigChange} placeholder="Ketik Judul Bab di sini..." className="w-full rounded border-indigo-200 p-2 text-sm border-2 border-indigo-300 focus:ring-2 focus:ring-indigo-500" />
                      )}
                    </div>
                    
                    <div className="grid grid-cols-2 gap-2">
                       <input type="text" name="tujuanBab" value={taskConfig.tujuanBab} onChange={handleTaskConfigChange} placeholder="Tujuan Bab..." className="w-full rounded border-indigo-200 p-2 text-sm" />
                       <select name="fokusLevel" value={taskConfig.fokusLevel} onChange={handleTaskConfigChange} className="w-full rounded border-indigo-200 p-2 text-sm">
                          <option value="Basic">Basic</option>
                          <option value="Intermediate">Intermediate</option>
                          <option value="Advanced">Advanced</option>
                       </select>
                    </div>
                  </div>
                </div>
              )}

              {/* --- OTHER MODES UI --- */}
              {activeMode === 'MODUL_PEMBELAJARAN' && (
                <div className="bg-emerald-50 p-4 rounded-lg border border-emerald-100 animate-in fade-in slide-in-from-bottom-2">
                  <h3 className="font-bold text-emerald-800 text-sm mb-3 flex items-center gap-2"><GraduationCap size={14}/> Detail Modul</h3>
                  <div className="space-y-3">
                    <input type="text" name="topikModul" value={taskConfig.topikModul} onChange={handleTaskConfigChange} placeholder="Topik Modul..." className="w-full rounded border-emerald-200 p-2 text-sm" />
                    <p className="text-xs text-emerald-600">AI akan membuat struktur: Learning Obj &rarr; Materi &rarr; Quiz &rarr; Tugas.</p>
                  </div>
                </div>
              )}

              {activeMode === 'TEMPLATE_TOOLBOX' && (
                <div className="bg-amber-50 p-4 rounded-lg border border-amber-100 animate-in fade-in slide-in-from-bottom-2">
                  <h3 className="font-bold text-amber-800 text-sm mb-3 flex items-center gap-2"><Wrench size={14}/> Detail Template/Alat</h3>
                  <div className="space-y-3">
                    <select name="jenisTemplate" value={taskConfig.jenisTemplate} onChange={handleTaskConfigChange} className="w-full rounded border-amber-200 p-2 text-sm">
                      <option value="SOP">SOP</option><option value="Formulir">Formulir</option><option value="Checklist">Checklist</option><option value="Worksheet">Worksheet</option><option value="Cheatsheet">Cheatsheet</option>
                    </select>
                    <input type="text" name="topikModul" value={taskConfig.topikModul} onChange={handleTaskConfigChange} placeholder="Topik spesifik template..." className="w-full rounded border-amber-200 p-2 text-sm" />
                  </div>
                </div>
              )}

              {(activeMode === 'SCRIPT_VIDEO' || activeMode === 'SCRIPT_AUDIO') && (
                <div className="bg-rose-50 p-4 rounded-lg border border-rose-100 animate-in fade-in slide-in-from-bottom-2">
                  <h3 className="font-bold text-rose-800 text-sm mb-3 flex items-center gap-2">{activeMode === 'SCRIPT_VIDEO' ? <Video size={14}/> : <Mic size={14}/>} Detail Script Multimedia</h3>
                  <div className="space-y-3">
                    <input type="text" name="judulScript" value={taskConfig.judulScript} onChange={handleTaskConfigChange} placeholder="Judul / Topik Script..." className="w-full rounded border-rose-200 p-2 text-sm" />
                    <input type="text" name="durasiScript" value={taskConfig.durasiScript} onChange={handleTaskConfigChange} placeholder="Estimasi Durasi..." className="w-full rounded border-rose-200 p-2 text-sm" />
                  </div>
                </div>
              )}

               {activeMode === 'PERPANJANG_PARAGRAF' && (
                <div className="bg-purple-50 p-4 rounded-lg border border-purple-100 animate-in fade-in slide-in-from-bottom-2">
                  <h3 className="font-semibold text-purple-800 text-sm mb-3">Input Teks Pendek</h3>
                  <textarea name="teksAwal" value={extendConfig.teksAwal} onChange={handleExtendChange} rows={4} placeholder="Paste paragraf..." className="w-full rounded border-purple-200 p-2 text-sm mb-2" />
                  <input type="text" name="targetPanjang" value={extendConfig.targetPanjang} onChange={handleExtendChange} placeholder="Target jumlah kata" className="w-full rounded border-purple-200 p-2 text-sm" />
                </div>
              )}
            </div>
          </div>

          <div className="flex-1 bg-slate-50 flex flex-col p-6 overflow-hidden">
             <div className="flex justify-between items-center mb-4">
                <h2 className="text-lg font-bold text-slate-700 flex items-center gap-2"><span className="w-6 h-6 rounded-full bg-slate-200 flex items-center justify-center text-xs text-slate-600">2</span> Generated Master Prompt</h2>
                <button onClick={copyToClipboard} className={`flex items-center gap-2 px-4 py-2 rounded-md text-sm font-medium transition-all ${isCopied ? 'bg-green-600 text-white hover:bg-green-700' : 'bg-slate-800 text-white hover:bg-slate-900'}`}>
                  {isCopied ? <Check size={16} /> : <Copy size={16} />}{isCopied ? 'Tersalin!' : 'Copy Prompt'}
                </button>
             </div>
             <div className="flex-1 relative shadow-inner">
                <textarea ref={textAreaRef} className="w-full h-full p-4 rounded-lg border border-slate-300 font-mono text-sm leading-relaxed text-slate-600 resize-none focus:ring-2 focus:ring-indigo-500 focus:outline-none" value={generatedPrompt} readOnly />
             </div>
             <div className="mt-4 p-3 bg-white border border-slate-200 rounded-md text-xs text-slate-500 flex justify-between items-center shadow-sm">
                <div className="flex items-center gap-2"><div className="bg-slate-100 p-1 rounded"><RefreshCw size={12}/></div><span>Gunakan "Quick Access AI" di atas untuk hasil terbaik.</span></div>
             </div>
          </div>
        </main>
      </div>
    </div>
  );
};
